in hardfloat_sys 

extern entity iNToRecFN<
    #uint intWidth,
    #uint expWidth,
    #uint sigWidth
>(
    control: uint<1>,
    signedIn: bool,
    in: uint<intWidth>,
    roundingMode: uint<3>,
    out: inv &uint<{expWidth + sigWidth + 1}>,
    exceptionFlags: inv &uint<5>
);

Description taken from the Berkeley Hardfloat documentation:

Converts from an integer type to floating-point in recoded form. The input named in is interpreted as an unsigned integer if signedIn is false, or as a signed integer if signedIn is true.
